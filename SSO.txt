#include<stdio.h>
#include<conio.h>
int main()
{
	int array[100],n,i,j,temp,min;
	clrscr();
	printf("Enter the Limit of Array");
	scanf("%d",&n);
	printf("Enter %d Elements:- \n",n);
	for(i=0;i<n;i++)
	{
		scanf("%d",&array[i]);
	}
	for(i=0;i<n-1;i++)
	{
	min=i;
	for(j=i+1;j<n;j++)
	{
	if(array[j]<array[min])
	{
	min=j;
	}
	}
	temp=array[i];
	array[i]=array[min];
	array[min]=temp;
	}
	printf("Sorted Result:-");
	for(i=0;i<n;i++)
	{
	printf("%d ",array[i]);
	}
	getch();
	return 0;
}
