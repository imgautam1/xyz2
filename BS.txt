#include<stdio.h>
#include<conio.h>

int main()
{
	int array[100],start,end,mid,n,i,search;
	printf("Enter the Number:");
	scanf("%d",&n);
	printf("Enter %d Integers =\n",n);
	for(i=0;i<n;i++)
	{
		scanf("%d",&array[i]);
	}
	printf("Enter the number You have to search =");
	scanf("%d",&search);
	start=0;
	end=n-1;
	mid=(start+end)/2; 
	while (start<=end)
	{
		if(array[mid]<search)
			start = mid+1;
		else if(array[mid]==search)
		{
			printf("Element %d is Located at %d",search,mid+1);
			break;
	 	}
	 	else 
	 	end = mid-1;
	 	mid=(start+end)/2;
	}
	if (start>end)
	 printf("Not found! %d is not present in the list.\n", search);
	return 0;
	
}
