#include <stdio.h>

long long factorialIterative(int n) {
    long long result = 1;
    if (n < 0) {
        return -1; 
    }
    for (int i = 1; i <= n; ++i) {
        result *= i;
    }
    return result;
}

int main() {
    int n;
    printf("Enter Number:-");
    scanf("%d",&n);
    printf("Factorial of %d is %lld\n", n, factorialIterative(n));
    return 0;
}
